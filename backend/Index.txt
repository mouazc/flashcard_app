<script>
    function loadFlashcard() {
        fetch('view_flashcard.php')
            .then(response => response.text())
            .then(data => {
                document.getElementById('flashcards-output').innerHTML = data;
            })
            .catch(error => console.error('Error loading flashcards:', error));
    }
</script>
